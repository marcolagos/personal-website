import React from "react";
import { motion } from "framer-motion";

const loadingJSON = require("../../../Data/Other/loading.json");

function Loading() {
	const blocks = loadingJSON.items.map((item, index) => {
		return (
			<motion.div
				key={index}
				animate={{
					rotate: item.animate.rotate,
					x: item.animate.x,
          y: item.animate.y,
          height: item.animate.height,
          width: item.animate.width,
          opacity: item.animate.opacity,
				}}
				transition={{
					repeat: Infinity,
					repeatType: "reverse",
					duration: 1.5,
					ease: "easeInOut",
				}}
				style={{
					background: "#00ff33",
					height: item.height,
					width: item.width,
					borderRadius: item.radius,
					position: "absolute",
				}}
			></motion.div>
		);
	});

	return (
		<div className="loading">
			{/* <span style={{ fontSize: 150, fontWeight: 400 }}>M</span> */}
			{blocks}
		</div>
	);
}
export default Loading;
